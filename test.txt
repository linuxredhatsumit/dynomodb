kind: ExternalSecret
metadata:
  name: coralogix-secret
  namespace: coralogix

  refreshInterval: 1h
  secretStoreRef:
    name: aws-secretstore
    kind: SecretStore
  target:
    name: coralogix-keys
    creationPolicy: Owner
  data:
    - secretKey: PRIVATE_KEY
      remoteRef:
        key: coralogix-dev-keys   # <-- AWS Secret name
        property: PRIVATE_KEY
